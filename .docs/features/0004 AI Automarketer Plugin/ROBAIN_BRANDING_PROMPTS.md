## Photoshop Automation Scripts

### Script 1: Batch Export Multi-Size Icons
**Purpose:** Export app icon in all required sizes

**File:** `export-icon-sizes.jsx`

```javascript
// Photoshop JSX Script: Export Icon in Multiple Sizes
// Save this as export-icon-sizes.jsx and run from File > Scripts > Browse

#target photoshop

// Define output sizes
var sizes = [1024, 512, 256, 128, 64, 32, 16];
var baseName = "robain-app-icon";

// Get the active document
var doc = app.activeDocument;

// Get the user's desktop for output
var outputFolder = Folder.selectDialog("Select output folder for icons");
if (!outputFolder) {
    alert("No folder selected. Script cancelled.");
} else {
    // Save original size first
    var originalWidth = doc.width.as("px");
    var originalHeight = doc.height.as("px");
    
    // Process each size
    for (var i = 0; i < sizes.length; i++) {
        var size = sizes[i];
        
        // Duplicate document
        var tempDoc = doc.duplicate(baseName + "-" + size + "px", true);
        
        // Resize
        tempDoc.resizeImage(
            UnitValue(size, "px"),
            UnitValue(size, "px"),
            null,
            ResampleMethod.BICUBIC
        );
        
        // Export as PNG
        var pngFile = new File(outputFolder + "/" + baseName + "-" + size + ".png");
        var pngOptions = new PNGSaveOptions();
        pngOptions.compression = 9;
        pngOptions.interlaced = false;
        
        tempDoc.saveAs(pngFile, pngOptions, true, Extension.LOWERCASE);
        
        // Close temp document
        tempDoc.close(SaveOptions.DONOTSAVECHANGES);
        
        // Progress
        app.activeDocument = doc; // Return focus to original
    }
    
    alert("Exported " + sizes.length + " icon sizes to:\n" + outputFolder.fsName);
}
```

**Usage:**
1. Open `robain-app-icon.psd` in Photoshop
2. Go to `File > Scripts > Browse`
3. Select `export-icon-sizes.jsx`
4. Choose output folder
5. Script will generate all icon sizes

### Script 2: Batch Export Character Variations
**Purpose:** Export all Robain character variations with consistent settings

**File:** `export-character-variations.jsx`

```javascript
// Photoshop JSX Script: Export All Character Variations
// Save this as export-character-variations.jsx

#target photoshop

// Define variations with their export specs
var variations = [
    {name: "robain-professional-main", width: 2048, height: 2048},
    {name: "robain-creative-banner", width: 1920, height: 1080},
    {name: "robain-analytical-data", width: 1600, height: 1200},
    {name: "robain-helpful-welcome", width: 1080, height: 1920},
    {name: "robain-celebrating-success", width: 1600, height: 1200},
    {name: "robain-loading-frame", width: 800, height: 800}
];

// Get output folder
var outputFolder = Folder.selectDialog("Select output folder for character exports");
if (!outputFolder) {
    alert("No folder selected. Script cancelled.");
} else {
    var doc = app.activeDocument;
    
    // Get base name from document name
    var docName = doc.name.replace(/\.[^\.]+$/, ''); // Remove extension
    
    // Export high-res version
    var pngFile = new File(outputFolder + "/" + docName + ".png");
    var pngOptions = new PNGSaveOptions();
    pngOptions.compression = 9;
    pngOptions.interlaced = false;
    
    doc.saveAs(pngFile, pngOptions, true, Extension.LOWERCASE);
    
    // Also export web-optimized version (50% size)
    var webDoc = doc.duplicate(docName + "-web", true);
    webDoc.resizeImage(
        UnitValue(doc.width.as("px") * 0.5, "px"),
        UnitValue(doc.height.as("px") * 0.5, "px"),
        null,
        ResampleMethod.BICUBIC
    );
    
    var webFile = new File(outputFolder + "/" + docName + "-web.png");
    webDoc.saveAs(webFile, pngOptions, true, Extension.LOWERCASE);
    webDoc.close(SaveOptions.DONOTSAVECHANGES);
    
    alert("Exported:\n" + pngFile.fsName + "\n" + webFile.fsName);
}
```

**Usage:**
1. Open any Robain character PSD file
2. Run script via `File > Scripts > Browse`
3. Exports full resolution + web-optimized version

### Script 3: Convert to Web-Safe Formats
**Purpose:** Convert PNGs to WebP and optimize file sizes

**File:** `convert-to-webp.sh` (Bash script for macOS)

```bash
#!/bin/bash
# Convert PNG images to WebP format for web optimization
# Requires: brew install webp

# Colors for output
GREEN='\033[0;32m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

echo -e "${BLUE}=== Robain Image Optimization ===${NC}"

# Check if cwebp is installed
if ! command -v cwebp &> /dev/null; then
    echo "Error: cwebp not found. Install with: brew install webp"
    exit 1
fi

# Input folder
if [ -z "$1" ]; then
    echo "Usage: ./convert-to-webp.sh /path/to/images"
    exit 1
fi

INPUT_DIR="$1"
OUTPUT_DIR="${INPUT_DIR}/webp"

# Create output directory
mkdir -p "$OUTPUT_DIR"

# Process each PNG file
COUNT=0
for file in "$INPUT_DIR"/*.png; do
    if [ -f "$file" ]; then
        filename=$(basename "$file" .png)
        
        # Convert to WebP with 90% quality
        cwebp -q 90 "$file" -o "$OUTPUT_DIR/${filename}.webp"
        
        # Get file sizes
        ORIGINAL_SIZE=$(stat -f%z "$file")
        WEBP_SIZE=$(stat -f%z "$OUTPUT_DIR/${filename}.webp")
        SAVINGS=$(echo "scale=1; (1 - $WEBP_SIZE / $ORIGINAL_SIZE) * 100" | bc)
        
        echo -e "${GREEN}âœ“${NC} ${filename}.png â†’ ${filename}.webp (${SAVINGS}% smaller)"
        COUNT=$((COUNT + 1))
    fi
done

echo -e "${BLUE}Converted ${COUNT} images to WebP${NC}"
echo -e "${BLUE}Output: ${OUTPUT_DIR}${NC}"
```

**Usage:**
```bash
chmod +x convert-to-webp.sh
./convert-to-webp.sh /path/to/exported/pngs
```

## AI Image Generation Tips

### Best Practices

1. **Be Specific:** Include exact colors, poses, emotions, and context
2. **Reference Style:** Mention "vector art," "modern illustration," or specific art styles
3. **Composition:** Specify "centered," "rule of thirds," "dynamic angle," etc.
4. **Resolution:** Always request "4K resolution" or "high resolution"
5. **Iterate:** Generate 4-8 variations, then pick the best and refine
6. **Consistency:** Reuse successful prompt structures for character consistency

### Refinement Prompts

If first generation isn't perfect, use these refinement prompts:

**Make more professional:**
```
Make the character more professional and business-appropriate, increase the polish and corporate aesthetic, sharper details, cleaner lines
```

**Make more friendly:**
```
Make the character more approachable and friendly, softer features, warmer expression, less formal pose, more inviting body language
```

**Improve color consistency:**
```
Adjust colors to exactly match this palette: primary blue #3B82F6, dark blue #1E40AF, light blue #93C5FD. Ensure consistent blue tones throughout.
```

**Simplify for icon:**
```
Simplify the design for use as a small icon, reduce fine details, increase contrast, make features bolder and more recognizable at small sizes
```

## Integration with Shopify App

### Code Examples

**React Component - Display Robain Avatar:**

```tsx
// components/RobainAvatar.tsx
import { useState, useEffect } from 'react';

interface RobainAvatarProps {
  variant: 'professional' | 'creative' | 'analytical' | 'helpful' | 'celebrating' | 'loading';
  size?: 'sm' | 'md' | 'lg' | 'xl';
  alt?: string;
}

const AVATAR_URLS = {
  professional: '/assets/robain-professional-main-web.webp',
  creative: '/assets/robain-creative-banner-web.webp',
  analytical: '/assets/robain-analytical-data-web.webp',
  helpful: '/assets/robain-helpful-welcome-web.webp',
  celebrating: '/assets/robain-celebrating-success-web.webp',
  loading: '/assets/robain-loading.webp',
};

const SIZE_CLASSES = {
  sm: 'w-16 h-16',
  md: 'w-24 h-24',
  lg: 'w-32 h-32',
  xl: 'w-48 h-48',
};

export function RobainAvatar({ variant, size = 'md', alt }: RobainAvatarProps) {
  const [isLoaded, setIsLoaded] = useState(false);

  return (
    <div className={`relative ${SIZE_CLASSES[size]} transition-opacity duration-300 ${isLoaded ? 'opacity-100' : 'opacity-0'}`}>
      <img
        src={AVATAR_URLS[variant]}
        alt={alt || `Robain - ${variant} mode`}
        className="w-full h-full object-contain"
        onLoad={() => setIsLoaded(true)}
        loading="lazy"
      />
      {variant === 'loading' && (
        <div className="absolute inset-0 flex items-center justify-center">
          <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-blue-500" />
        </div>
      )}
    </div>
  );
}
```

**Usage:**
```tsx
<RobainAvatar variant="professional" size="lg" />
<RobainAvatar variant="loading" size="md" />
<RobainAvatar variant="celebrating" size="xl" alt="Campaign successful!" />
```
